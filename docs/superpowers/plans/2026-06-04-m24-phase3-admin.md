# M24 Web UI Phase 3 — manage (admin) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** CSRF-protected settings forms in the web UI wrapping the existing stores: self-service (tokens/SSH keys/password), repo settings (public/grants/deploy keys/webhooks/policy) gated by the repo-level `admin` perm, and instance admin (users/repos/quotas/hooks) gated by `IsAdmin`.

**Architecture:** Approach A from the spec (`docs/superpowers/specs/2026-06-04-m24-phase3-admin-design.md`): widen `web.DataStore` with admin methods (still satisfied by `*sqlitestore.Store` through the cmd adapter), add small consumer-declared interfaces to `web.Deps` for webhooks/policy/hooks/quotas/repo-init, POST-redirect-GET forms with one handler per action, web handlers emit audit events for every state change.

**Tech Stack:** Go stdlib `net/http` + `html/template` (existing chassis), sqlite via existing `sqlitestore`, no new dependencies.

**Branch:** `m24-phase3-admin`. **Review protocol:** per-task spec + code-quality reviews (subagent-driven), then `roborev-refine` on the branch at the end (project M1+ convention).

**Conventions that bind every task:**
- TDD: write the failing test first, watch it fail, implement, watch it pass, commit.
- Every state-changing handler: method check → `ParseForm` → `checkCSRF` → authorize → validate → store call → audit emit → metric emit → flash + 303 (or direct secret render).
- Authz failure = uniform 404 via `s.renderError(w, r, http.StatusNotFound, "not found")`. CSRF failure = 403. Anonymous on settings pages = 302 to `/login?next=<path>`.
- Audit events carry `slog.String("actor", sess.Name)`, `slog.String("source", "web")`, plus target attrs.
- Run `gofmt -l` and `go test ./internal/web/ ./internal/auth/sqlitestore/` before every commit.

```go
// internal/web/flash.go
package web

import (
	"encoding/base64"
	"net/http"
)

const flashCookieName = "bvcs_flash"

// setFlash stores a short one-shot notice shown on the next page render.
// Base64url-encoded so arbitrary text survives the cookie value charset.
func setFlash(w http.ResponseWriter, msg string, secure bool) {
	http.SetCookie(w, &http.Cookie{
		Name:     flashCookieName,
		Value:    base64.RawURLEncoding.EncodeToString([]byte(msg)),
		Path:     "/",
		HttpOnly: true,
		Secure:   secure,
		SameSite: http.SameSiteLaxMode,
		MaxAge:   60,
	})
}

// takeFlash returns the pending notice ("" if none) and clears the cookie.
func takeFlash(w http.ResponseWriter, r *http.Request) string {
	c, err := r.Cookie(flashCookieName)
	if err != nil || c.Value == "" {
		return ""
	}
	http.SetCookie(w, &http.Cookie{
		Name: flashCookieName, Value: "", Path: "/", MaxAge: -1, HttpOnly: true,
	})
	b, err := base64.RawURLEncoding.DecodeString(c.Value)
	if err != nil {
		return ""
	}
	return string(b)
}
```

```go
// internal/web/authz.go
package web

import (
	"net/http"
	"net/url"

	"github.com/bucketvcs/bucketvcs/internal/auth"
)

// requireUser ensures a logged-in session; on failure it 303s to login with
// ?next= and returns false.
func (s *server) requireUser(w http.ResponseWriter, r *http.Request) bool {
	if SessionFromContext(r.Context()) != nil {
		return true
	}
	http.Redirect(w, r, "/login?next="+url.QueryEscape(r.URL.Path), http.StatusSeeOther)
	return false
}

// isGlobalAdmin reports whether the request's session is a global admin.
func isGlobalAdmin(r *http.Request) bool {
	sess := SessionFromContext(r.Context())
	return sess != nil && sess.IsAdmin
}

// canAdminRepo reports whether the session may manage (tenant, repo) settings:
// global admin, or holder of the repo-level admin permission. Errors (including
// no-such-repo) deny — callers respond with uniform 404.
func (s *server) canAdminRepo(r *http.Request, tenant, repo string) bool {
	sess := SessionFromContext(r.Context())
	if sess == nil {
		return false
	}
	if sess.IsAdmin {
		return true
	}
	perm, err := s.store.LookupRepoPerm(r.Context(), actorFromSession(sess), tenant, repo)
	return err == nil && perm == auth.PermAdmin
}
```

```go
// assertFormSecurity runs the standard matrix against a state-changing route:
// anon GET→303 login (when get=true), POST without CSRF→403, POST as
// insufficient role→404. h is the full handler from newTestHandler-style setup.
type secOpts struct {
	path      string        // e.g. "/acme/demo/settings/public"
	form      url.Values    // valid form body (without csrf_token)
	asSession *auth.Session // a session that must be DENIED with 404 (nil to skip)
}

func assertFormSecurity(t *testing.T, h http.Handler, o secOpts) {
	t.Helper()
	// 1. anonymous POST (no session cookie) → 303 to /login
	r := httptest.NewRequest(http.MethodPost, o.path, strings.NewReader(o.form.Encode()))
	r.Header.Set("Content-Type", "application/x-www-form-urlencoded")
	rec := httptest.NewRecorder()
	h.ServeHTTP(rec, r)
	if rec.Code != http.StatusSeeOther {
		t.Fatalf("anon POST %s: code = %d, want 303", o.path, rec.Code)
	}
	// 2. authenticated POST without CSRF → 403
	r = httptest.NewRequest(http.MethodPost, o.path, strings.NewReader(o.form.Encode()))
	r.Header.Set("Content-Type", "application/x-www-form-urlencoded")
	attachSession(r, adminSession) // helper + fixture defined alongside; see note below
	rec = httptest.NewRecorder()
	h.ServeHTTP(rec, r)
	if rec.Code != http.StatusForbidden {
		t.Fatalf("no-CSRF POST %s: code = %d, want 403", o.path, rec.Code)
	}
	// 3. insufficient role with valid CSRF → 404
	if o.asSession != nil {
		r = csrfPost(t, o.path, o.form) // builds matching cookie+field
		attachSession(r, o.asSession)
		rec = httptest.NewRecorder()
		h.ServeHTTP(rec, r)
		if rec.Code != http.StatusNotFound {
			t.Fatalf("denied-role POST %s: code = %d, want 404", o.path, rec.Code)
		}
	}
}
```

```go
// renderBuffered renders a page to a buffer first so template errors yield a
// clean 500 instead of a half-written 200.
func (s *server) renderBuffered(w http.ResponseWriter, page string, data any) error {
	var buf bytes.Buffer
	if err := s.render.render(&buf, page, data); err != nil {
		s.logger.Error("web: render", "page", page, "err", err)
		return err
	}
	w.Header().Set("Content-Type", "text/html; charset=utf-8")
	_, _ = w.Write(buf.Bytes())
	return nil
}
```
