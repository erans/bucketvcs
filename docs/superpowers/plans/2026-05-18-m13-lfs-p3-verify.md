## Phase 3 design decisions

1. **Verify path: `POST /<tenant>/<repo>.git/info/lfs/objects/<oid>/verify`** per LFS spec. The `<oid>` is in the URL path, NOT in the body. The body is `{"oid":"...","size":N}` (a `VerifyRequest` already defined in P0's `types.go`). The handler verifies that the URL OID matches the body OID and that `Head(repoLFSPrefix + oid).Size == size`.
2. **RequiredAction = ActionWrite.** Verify is only meaningful in the context of an upload. ParseRoute returns ActionWrite for OpLFSVerify; RunAuth enforces. No secondary check needed inside the handler for verify (unlike batch, which needs the upload/download branch).
3. **Same `lfs.NewHTTPHandler` for batch + verify.** The handler internally dispatches on URL path: `splitLFSPath` (extended) returns one of `lfsRouteBatch` / `lfsRouteVerify` / not-LFS. The two sub-handlers share auth/logger/store/dependencies.
4. **Status code map (per spec §6):**
   - 200: size matches, object present.
   - 422: size mismatch (stored != claimed).
   - 404: object missing in storage (the client uploaded incorrectly or storage was wiped).
   - 500: backend error.
5. **No OID-in-path/body mismatch enforcement at the URL layer.** If a client crafts a path OID different from the body OID, return 422 with a clear message — but this is a client bug, not a security issue (the URL is already authed, the body adds no new authority).
6. **Smoke script: localfs only.** No MinIO dependency; the script runs entirely against the in-process gateway with a localfs backend. If `git-lfs` is not on PATH, the script `exit 77` (autoconf SKIP convention) so CI can choose how to handle it.
7. **No new Op-keyed gateway metric.** The handler emits per-verify metrics+audit; the gateway-level routing layer doesn't need a separate counter.

#### Step 2: Implement verify.go

Create `internal/lfs/verify.go`:

package lfs

import (
    "context"
    "errors"
    "fmt"
)

// ErrVerifySizeMismatch is returned by Verify when the stored object
// exists but its size does not match the claimed size. The handler
// maps it to HTTP 422.
var ErrVerifySizeMismatch = errors.New("lfs: verify size mismatch")

// ErrVerifyNotFound is returned by Verify when the stored object is
// absent. The handler maps it to HTTP 404.
var ErrVerifyNotFound = errors.New("lfs: verify object not found")

// Verify confirms that an object with the given oid exists in the
// store and has the claimed size. Returns:
//
//   - nil: object exists and size matches.
//   - ErrVerifySizeMismatch: object exists with a different size.
//   - ErrVerifyNotFound: object is missing.
//   - other error: backend failure.
//
// Verify is pure logic over Store.Head; no side effects, no HTTP.
func Verify(ctx context.Context, store *Store, oid string, claimedSize int64) error {
    size, exists, err := store.Head(ctx, oid)
    if err != nil {
        return fmt.Errorf("lfs.Verify: head: %w", err)
    }
    if !exists {
        return ErrVerifyNotFound
    }
    if size != claimedSize {
        return fmt.Errorf("%w: stored=%d, claimed=%d", ErrVerifySizeMismatch, size, claimedSize)
    }
    return nil
}
```

type lfsRoute int

const (
    lfsRouteNone lfsRoute = iota
    lfsRouteBatch
    lfsRouteVerify
)

// parseLFSPath parses /<tenant>/<repo>.git/info/lfs/objects/<rest>
// and returns the sub-route plus extracted components. For
// lfsRouteBatch, oid is empty. For lfsRouteVerify, oid is set.
func parseLFSPath(p string) (tenant, repo, oid string, route lfsRoute) {
    if p != path.Clean(p) {
        return "", "", "", lfsRouteNone
    }
    parts := strings.SplitN(strings.TrimPrefix(p, "/"), "/", 3)
    if len(parts) < 3 {
        return "", "", "", lfsRouteNone
    }
    tenant = parts[0]
    repoSeg := parts[1]
    rest := parts[2]
    if !strings.HasSuffix(repoSeg, ".git") || repoSeg == ".git" {
        return "", "", "", lfsRouteNone
    }
    repo = strings.TrimSuffix(repoSeg, ".git")
    if !validRouteName(tenant) || !validRouteName(repo) {
        return "", "", "", lfsRouteNone
    }
    switch {
    case rest == "info/lfs/objects/batch":
        return tenant, repo, "", lfsRouteBatch
    case strings.HasPrefix(rest, "info/lfs/objects/") && strings.HasSuffix(rest, "/verify"):
        oid = strings.TrimSuffix(strings.TrimPrefix(rest, "info/lfs/objects/"), "/verify")
        if !validOID.MatchString(oid) {
            return "", "", "", lfsRouteNone
        }
        return tenant, repo, oid, lfsRouteVerify
    }
    return "", "", "", lfsRouteNone
}
```

```go
func handleVerify(ctx context.Context, w http.ResponseWriter, r *http.Request, deps Deps, tenant, repo, oid, user string) {
    logger := deps.Logger
    if logger == nil {
        logger = slog.Default()
    }
    body := http.MaxBytesReader(w, r.Body, 64*1024) // verify body is tiny
    var vreq VerifyRequest
    if err := json.NewDecoder(body).Decode(&vreq); err != nil {
        WriteError(w, http.StatusUnprocessableEntity, "unprocessable: "+err.Error())
        emitVerifyRequestMetric(ctx, logger, "error")
        return
    }
    if vreq.OID != oid {
        WriteError(w, http.StatusUnprocessableEntity, "body oid does not match URL oid")
        emitVerifyRequestMetric(ctx, logger, "error")
        return
    }
    store := deps.NewStore(tenant, repo)
    err := Verify(ctx, store, oid, vreq.Size)
    switch {
    case err == nil:
        w.WriteHeader(http.StatusOK)
        emitVerifyRequestMetric(ctx, logger, "ok")
        emitLFSVerify(ctx, logger, tenant+"/"+repo, user, oid, vreq.Size, "ok")
    case errors.Is(err, ErrVerifyNotFound):
        WriteError(w, http.StatusNotFound, "object not uploaded")
        emitVerifyRequestMetric(ctx, logger, "missing")
        emitLFSVerify(ctx, logger, tenant+"/"+repo, user, oid, vreq.Size, "missing")
    case errors.Is(err, ErrVerifySizeMismatch):
        WriteError(w, http.StatusUnprocessableEntity, err.Error())
        emitVerifyRequestMetric(ctx, logger, "size_mismatch")
        emitLFSVerify(ctx, logger, tenant+"/"+repo, user, oid, vreq.Size, "size_mismatch")
    default:
        WriteError(w, http.StatusInternalServerError, "internal error")
        emitVerifyRequestMetric(ctx, logger, "error")
        emitLFSVerify(ctx, logger, tenant+"/"+repo, user, oid, vreq.Size, "error")
    }
}
```

The handler's main body needs to compute `user` once (from actor) and pass it to either handleBatch or handleVerify. Refactor accordingly.

// emitLFSVerify records a "lfs.verify" audit event after a verify
// request completes. Matches the M11 audit flat-attrs shape.
//
// repo is "<tenant>/<repo>". user is the actor name or "" for
// anonymous (impossible today since verify is ActionWrite-only, but
// the param keeps the function shape consistent with emitLFSBatch).
// oid is the verified object's hex SHA-256. size is the claimed size
// (the body's reported size — may differ from stored size on the
// size_mismatch path).
func emitLFSVerify(ctx context.Context, logger *slog.Logger, repo, user, oid string, size int64, result string) {

        slog.String("event", "lfs.verify"),
        slog.String("repo", repo),
        slog.String("user", user),
        slog.String("oid", oid),
        slog.Int64("size", size),
        slog.String("result", result),
    )
}

# scripts/m13-lfs-smoke-local.sh

# End-to-end smoke test for M13 LFS against a localfs gateway. Drives

# Skips with exit 77 if git-lfs is not installed (autoconf SKIP
# convention). Requires `bucketvcs` and `git` on PATH.

set -euo pipefail

if ! command -v git-lfs >/dev/null 2>&1; then
    echo "SKIP: git-lfs not on PATH; install from https://git-lfs.com/"
    exit 77
fi

ROOT="$(mktemp -d)"
echo "smoke root: $ROOT"

cleanup_status=0
cleanup() {
    cleanup_status=$?
    if [[ $cleanup_status -ne 0 ]]; then
        echo "==== SMOKE FAILED (status=$cleanup_status) ===="
        echo "Preserved root for forensics: $ROOT"
        if [[ -f "$ROOT/serve.log" ]]; then
            echo "==== serve.log (last 50 lines) ===="
            tail -50 "$ROOT/serve.log" || true
        fi
        if [[ -n "${SERVE_PID:-}" ]]; then
            kill "$SERVE_PID" 2>/dev/null || true
        fi
        return
    fi
    [[ -n "${SERVE_PID:-}" ]] && kill "$SERVE_PID" 2>/dev/null || true
    rm -rf "$ROOT"
    echo "M13_SMOKE_OK"
}
trap cleanup EXIT

# 1. Build bucketvcs (assume repo root is the script's parent's parent).
REPO_ROOT="$(cd "$(dirname "$0")/.." && pwd)"
BUCKETVCS="$ROOT/bucketvcs"
( cd "$REPO_ROOT" && go build -o "$BUCKETVCS" ./cmd/bucketvcs )

# 2. Set up auth + signing key.
STORE_DIR="$ROOT/store"
mkdir -p "$STORE_DIR"
AUTH_DB="$ROOT/auth.db"
SIGNING_KEY_FILE="$ROOT/signing.key"
openssl rand -hex 16 > "$SIGNING_KEY_FILE"

PORT=$(python3 -c 'import socket; s=socket.socket(); s.bind(("127.0.0.1",0)); print(s.getsockname()[1]); s.close()')
BASE_URL="http://127.0.0.1:$PORT"

"$BUCKETVCS" user create alice --auth-db "$AUTH_DB"
TOKEN=$("$BUCKETVCS" token create alice --auth-db "$AUTH_DB" | tail -1)
"$BUCKETVCS" repo create acme/lfs-test --auth-db "$AUTH_DB"

    --addr "127.0.0.1:$PORT" \
    --store "localfs:$STORE_DIR" \
    --auth-db "$AUTH_DB" \
    --lfs \
    --proxied-url-signing-key "$SIGNING_KEY_FILE" \
    --proxied-url-base "$BASE_URL" \

SERVE_PID=$!

# 4. Wait for the server to be ready.
for i in $(seq 1 50); do

    fi
    sleep 0.1
done
curl -sf "$BASE_URL/healthz" >/dev/null || { echo "server did not start"; exit 1; }
echo "READY"

# 5. Create a local repo with an LFS-tracked file.
REPO="$ROOT/repo"
mkdir -p "$REPO"
( cd "$REPO" && \
    git init -q && \
    git lfs install --local && \
    git lfs track "*.bin" && \
    git config user.email smoke@example.com && \
    git config user.name smoke && \

    git add .gitattributes big.bin && \
    git commit -qm "add lfs object" && \

# 6. Confirm batch+verify markers in the server log.

grep -q '"event":"lfs.object.served"' "$ROOT/serve.log" || { echo "missing lfs.object.served event"; exit 1; }

echo "LFS_SMOKE_PUSH_OK"

( cd "$CLONE" && git lfs pull )
test -s "$CLONE/big.bin" || { echo "cloned LFS object is empty"; exit 1; }
cmp "$REPO/big.bin" "$CLONE/big.bin" || { echo "cloned LFS bytes differ"; exit 1; }
echo "LFS_SMOKE_PULL_OK"
